async function userLogIn(){
    logInButton.disabled = true;
    users.push({
        email: email.value,
        password: password.value,
    });

    await setItem('users', JSON.stringify(users));
    resetForm();
}


function resetForm(){
    email.value = '';
    password.value = '';
    logInButton.disabled = false;
}
